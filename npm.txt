# nếu chưa tạo nest cli chạy lệnh này
npm i -g @nestjs/cli

# tạo project
nest new name_project

# vào project
cd name_project

# hot reload
npm i --save-dev webpack-node-externals run-script-webpack-plugin webpack 

# Cài đặt TypeORM và Driver cho PostgreSQL
npm install @nestjs/typeorm typeorm pg

# tạo nhanh cấu trúc CRUD users
nest g resource users

# cài đặt quản lý cấu hình
npm i @nestjs/config

# thư viện xác thực trường dữ liệu isBlank, isEmail...
npm install class-validator class-transformer

# tạo module auth
nest g module auth
nest g service auth
nest g controller auth

#cài jwt
npm install --save @nestjs/jwt

# cài cache
npm install @nestjs/cache-manager cache-manager cache-manager-redis-store
npm install redis --save

# cài microservices và thư viện kết nối RabbitMQ và thư viện tự động kết nối lại nếu RabbitMQ bị ngắt quãng
npm install @nestjs/microservices amqplib amqp-connection-manager

#package.json
"start:dev": "nest build --webpack --webpackPath webpack-hmr.config.js --watch", => dùng cái này nhanh nhưng khó cho docker
"start:dev": "nest build --webpack --webpackPath webpack-hmr.config.js --watch", => dùng cái này phổ biến chậm hơn nhưng ổn định docker

// rate limiting 
npm i --save @nestjs/throttler
// kết hợp với redis
npm install --save @nest-lab/throttler-storage-redis ioredis

# lấy file backup từ pg local
pg_dump -U postgres -d user_crud > backup.sql

# import vào trong docker
docker exec -i nestjs-db psql -U postgres -d user_crud < backup.sql

#import dữ liệu vào docker
Get-Content backup.sql | docker exec -i nestjs-db psql -U postgres -d user_crud

# vào pg docker để kiểm tra dữ liệu
docker exec -it nestjs-db psql -U postgres -d user_crud
